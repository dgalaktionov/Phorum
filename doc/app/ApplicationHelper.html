  if topic.posts.empty?
    [topic.category, topic]
  else
    get_post_path(topic.posts.order(updated_at: :desc).first)
  end

    When a user deletes its own account, this is called to edit all its topics and posts

    # File app/helpers/application_helper.rb, line 43
def kill_user(user)
  user.posts.each do |p| 
    p.message = "Deleted message"
    p.save!
  end
  user.topics.each do |t| 
    t.message = "Deleted message"
    t.save!
  end
  user.destroy
end

def phorum_transform(text)
  auto_link(phorum_sanitize(text).gsub(/(?<img>(https?:\/\/)?[-a-zA-Z0-9.]+\/[-_a-zA-Z0-9&.\/]+\.(?i)(bmp|jpg|jpeg|png|gif|tif|exf|svg|wfm))/, '<a href="\k<img>"><img alt="Phorum Image" src="\k<img>" class="phorum-image"></img></a>'))
end
